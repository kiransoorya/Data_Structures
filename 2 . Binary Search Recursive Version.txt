#include<iostream>#include<math.h>using namespace std;bool recursive_binary_search(float array[],int first , int last,float target,int &location){		if(first>last){		location =-1;		return false;	}	else{		int mid=floor((first+last)/2);		if(target<array[mid]){			location=mid;			return recursive_binary_search(array,first,mid-1,target,location);		}		if(target>array[mid]){			location=mid;			return recursive_binary_search(array,mid+1,last,target,location);		}		if(target==array[mid]){			location=mid;			return true;		}	}}int main(){	int number_of_elements;	cout<<"ENTER THE NUMBER OF ELEMENTS IN THE ARRAY :";	cin>>number_of_elements;	float array[number_of_elements];	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<"ENTER VALUE NO. "<<element_index+1<<"\t";		cin>>array[element_index];		}	float target;	cout<<"ENTER THE ELEMENT TO BE FOUND :";	cin>>target;	int location;	int first=0;	bool found = recursive_binary_search(array,first,number_of_elements-1,target,location);	if(found==true)		cout<<"THE ELEMENT WAS FOUND AT :"<<location;	else		cout<<"THE ELEMENT WASN'T FOUND IN THE ARRAY. "<<location;	return 0;}
