#include<iostream>using namespace std;void exchange(float array[],int walker, int walker_minus_one){	float temp = array[walker];		array[walker]=array[walker_minus_one];	array[walker_minus_one]=temp;}void bubbleSort(float array[],int last){	int current = 0;	bool sorted = false;	while(current<last && sorted==false){		int walker = last;		sorted = true;		while(walker>current){			if(array[walker]<array[walker-1]){				sorted = false;							exchange(array, walker,walker-1);			}			walker-=1;				}		current+=1;		}}int main(){	int number_of_elements;	cout<<"ENTER THE NUMBER OF ELEMENTS IN THE ARRAY:";	cin>>number_of_elements;	float array[number_of_elements];	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<"Enter value no."<<element_index+1;		cin>>array[element_index];		}	cout<<"\nBefore Sorting\n";	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<array[element_index]<<" ";	}	bubbleSort(array,number_of_elements-1);	cout<<"\nAfter Sorting\n";	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<array[element_index]<<" ";	}	return 0;}
