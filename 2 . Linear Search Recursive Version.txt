#include <iostream>using namespace std;bool recursive_linear_search(float array[],int index,int last,int &location, float target){	if(index>last){		location =-1;		return false;	}	if(target==array[index]){		location = index;		return true;	}	return recursive_linear_search(array,index+1,last,location,target);}int main(){	int number_of_elements;	cout<<"ENTER THE NUMBER OF ELEMENTS FOR THE ARRAY :";	cin>>number_of_elements;	float array[number_of_elements];	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<"ENTER VALUES:";				cin>>array[element_index];		}	float target;	cout<<"ENTER THE ELEMENT TO BE FOUND :";	cin>>target;	int location;	int index=0;	bool found = recursive_linear_search(array,index,number_of_elements-1,location,target);	if(found==true)		cout<<"THE ELEMENT WAS FOUND IN :"<<location;	else		cout<<"THE ELEMENT WASN'T FOUND IN ARRAY . "<<location;	return 0;}
