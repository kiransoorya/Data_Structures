#include<iostream>#include <math.h>using namespace std;bool binary_search(float array[] ,int last ,float target ,int &location){	int begin=0;	int end=last;	bool is_present=false;	int mid;	while(begin<=last){		mid = floor((begin+end)/2);		if(target>array[mid])			begin=mid+1;		else if(target<array[mid])			end=mid-1;		else			begin=end+1;	}	location=mid;	if(target==array[location])		is_present=true;	return is_present;}int main(){	int number_of_elements;	cout<<"ENTER THE NUMBER OF ELEMENTS IN THE ARRAY :";	cin>>number_of_elements;	float array[number_of_elements];	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<"ENTER VALUE NO. "<<element_index+1<<"\t";		cin>>array[element_index];		}	float target;	cout<<"ENTER THE ELEMENT TO BE FOUND :";	cin>>target;	int location;	bool found = binary_search(array,number_of_elements-1,target,location);	if(found==true)		cout<<"THE ELEMENT WAS FOUND AT :"<<location;	else		cout<<"THE ELEMENT WASN'T FOUND IN THE ARRAY. "<<location;	return 0;}
