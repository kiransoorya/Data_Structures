#include<iostream>using namespace std;void InsertionSort(float array[],int last){	int current = 1;	while(current<=last){		float key =array[current];		int walker = current - 1;		while(walker>=0 && key<array[walker]){			array[walker+1]=array[walker];			walker-=1;				}		array[walker+1]=key;		current+=1;		}}int main(){	int number_of_elements;	cout<<"ENTER THE NUMBER OF ELEMENTS IN THE ARRAY:";	cin>>number_of_elements;	float array[number_of_elements];	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<"Enter value no."<<element_index+1;		cin>>array[element_index];		}	cout<<"\nBefore Sorting\n";	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<array[element_index]<<" ";	}	InsertionSort(array,number_of_elements-1);	cout<<"\nAfter Sorting\n";	for(int element_index=0;element_index<number_of_elements;element_index++){		cout<<array[element_index]<<" ";	}	return 0;}
